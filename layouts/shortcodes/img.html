{{- $src := .Get "src" -}} {{- $image := .Page.Resources.GetMatch $src -}} {{- if and (not $image) .Page.File -}} {{- $image = resources.Get (path.Join .Page.File.Dir $src) -}} {{- end -}} {{- $image_sizes := slice 200 400 600 800 1200 1920 -}} {{- if hugo.IsProduction -}} {{/* ============================================================ Mode production : conversion WebP et redimensionnement local via le pipeline d'images Hugo ============================================================ */}} {{- with $image -}} {{- $image_width := $image.Width -}} {{- $image_height := $image.Height -}} {{- if strings.Contains $src "_2x" -}} {{- $image_width = math.Floor (math.Div $image_width 2) -}} {{- $image_height = math.Floor (math.Div $image_height 2) -}} {{- end -}} {{- $srcset_parts := slice -}} {{- range $image_sizes -}} {{- if le . $image.Width -}} {{- $resized := $image.Resize (printf "%dx webp" .) -}} {{- $srcset_parts = $srcset_parts | append (printf "%s %dw" $resized.RelPermalink .) -}} {{- end -}} {{- end -}} {{- $fallback := $image.Resize "1000x webp" -}} {{- $lightbox := $image.Process "webp" -}} {{- end -}} {{- else -}} {{/* ============================================================ Mode développement : images telles quelles, sans traitement ============================================================ */}} {{- with $image -}} {{- $image_width := $image.Width -}} {{- $image_height := $image.Height -}} {{- if strings.Contains $src "_2x" -}} {{- $image_width = math.Floor (math.Div $image_width 2) -}} {{- $image_height = math.Floor (math.Div $image_height 2) -}} {{- end -}} {{- end -}} {{- end -}}
